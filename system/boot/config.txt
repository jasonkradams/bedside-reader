# Enable required hardware interfaces for the amp (and, later, the SPI screen)
dtparam=spi=on
dtparam=i2s=on

# Disable onboard audio (using MAX98357A via I2S instead)
dtparam=audio=off

# Automatically load overlays for detected cameras and DSI displays
camera_auto_detect=1
display_auto_detect=1

# Automatically load initramfs files, if found
auto_initramfs=1

# Run in 64-bit mode
arm_64bit=1

# Disable compensation for displays with overscan
disable_overscan=1

# Run as fast as firmware / board allows
arm_boost=1

# --- Appliance-Specific Overlays ---
# These are applied by the Raspberry Pi firmware from overlays/ on this FAT
# partition (U-Boot ignores the NixOS FDTDIR on this board, so hardware.deviceTree
# never reaches the kernel — the firmware owns the device tree here).
# IMPORTANT: enable overlays ONE AT A TIME. Applying all of them at once hung early
# boot. Audio is validated first; display/backlight/KMS stay commented until each
# is proven to boot on its own.

# MAX98357A I2S DAC/amp — snd-soc-max98357a + simple-audio-card (both in-kernel).
# sdmode-pin=26: the amp's SD/enable is wired to GPIO26 (phys 37) on this build
# (High=awake, Low=mute). The overlay's default sdmode-pin is GPIO4 — which on this
# build is the rotary encoder's A line, so the default would drive an input pin and
# fight the encoder. Pinning it to 26 both enables the amp and frees GPIO4.
dtoverlay=max98357a,sdmode-pin=26

# Display HAT Mini ST7789 on SPI0/CE1 (DC=GPIO9, 60MHz), validated via modetest.
# The panel-mipi-dbi driver loads its init sequence from a firmware file named
# "<first-compatible>.bin". Leaving the overlay's default compatible ("panel") makes
# it load panel.bin (shipped via hardware.firmware). Do NOT add
# dtparam=compatible=sitronix,st7789v — that makes it look for sitronix,st7789v.bin,
# which doesn't exist, so the panel never initialises (blank backlit screen).
dtoverlay=mipi-dbi-spi,spi0-1,speed=60000000
dtparam=width=320,height=240,width-mm=43,height-mm=27
dtparam=dc-gpio=9

# KMS/HDMI (vc4-kms-v3d) and gpio-backlight stay off: the app renders to the SPI
# panel's own DRM device and drives the GPIO13 backlight via sysfs itself.
#dtoverlay=vc4-kms-v3d
#max_framebuffers=2
#disable_fw_kms_setup=1
#dtoverlay=gpio-backlight,gpio=13,active_low=0

# Hardware watchdog (Zero 2 W has BCM2837 watchdog)
dtparam=watchdog=on

# --- Hardware Filters (Ignored on Pi Zero 2 W, safe to keep) ---
[cm4]
otg_mode=1

[cm5]
dtoverlay=dwc2,dr_mode=host

[pi5]
dtoverlay=nospi10
